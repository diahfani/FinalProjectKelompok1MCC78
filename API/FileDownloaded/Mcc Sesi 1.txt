create DATABASE db_hr_mcc

use db_hr_mcc

create table regions(

id int IDENTITY PRIMARY KEY not null,
name varchar (25) not null)

select * from regions

2.
create table country(

id varchar(2) primary key,
name varchar(40) not null,
region int not null)
ALTER TABLE country ADD FOREIGN KEY (region) REFERENCES regions(id)

select * from country

3.
create table locations(

id int PRIMARY KEY,
street_address varchar(40) null,
postal_code varchar(12) null,
city varchar(30) not null,
state_province varchar(40) null,
countrys varchar(2) not null)
ALTER TABLE locations ADD FOREIGN KEY (countrys) REFERENCES country(id)

select * from locations

drop table locations

4.
create table department(

id int primary key,
name varchar(30) not null,
location int not null,

manager int)
ALTER TABLE department ADD FOREIGN KEY (location) REFERENCES locations(id);
ALTER TABLE department ADD FOREIGN KEY (manager) REFERENCES employee(id);

select * from department

5.
create table employee(

id int primary key,
first_name varchar(20) not null,
last_name varchar(25) null,
email varchar(25) not null UNIQUE,
phone_number varchar(20) null,
hire_date date not null,
salary int null,
commision_pct float null,
manager int null,
jobs varchar(10)not null,
department int not null);

ALTER TABLE employee ADD FOREIGN KEY (manager) REFERENCES employee(id)
ALTER TABLE employee ADD FOREIGN KEY (jobs) REFERENCES job(id)
ALTER TABLE employee ADD FOREIGN KEY (department) REFERENCES department(id)

select * from employee

6.
create table job(

id varchar(10) PRIMARY KEY,
title varchar(35)not null,
min_salary int null,
max_salary int null)

select * from job

7.
create table history(

start_date date not null,
end_date date not null,
department_id int not null,
employee_id int not null,
job_id varchar(10) not null)

ALTER TABLE history ADD FOREIGN KEY (department_id) REFERENCES department(id);
ALTER TABLE history ADD FOREIGN KEY (employee_id) REFERENCES employee(id);
ALTER TABLE history ADD FOREIGN KEY (job_id) REFERENCES job(id);
